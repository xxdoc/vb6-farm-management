SELECT II.* FROM IMPORT_ITEM II
LEFT OUTER JOIN INVENTORY_DOC IVD ON (II.INVENTORY_DOC_ID = IVD.INVENTORY_DOC_ID)
WHERE (II.TRANSACTION_SEQ < 3559)
AND (IVD.DOCUMENT_DATE <= '2005-04-01 00:00:00')
ORDER BY II.TRANSACTION_SEQ DESC;

SELECT EI.* FROM EXPORT_ITEM EI
LEFT OUTER JOIN INVENTORY_DOC IVD ON (EI.INVENTORY_DOC_ID = IVD.INVENTORY_DOC_ID)
WHERE (EI.TRANSACTION_SEQ < 3559)
AND (IVD.DOCUMENT_DATE <= '2005-04-01 00:00:00')
ORDER BY EI.TRANSACTION_SEQ DESC;
