CREATE TABLE PARAMETERS
(
   PARAM_ID              ID_TYPE NOT NULL,
   PARAM_NO              CODE_TYPE,
   PARAM_DATE            DATE_TYPE,
   PARAM_AREA            ID_TYPE,
   FROM_SALE_DATE        DATE_TYPE,
   TO_SALE_DATE          DATE_TYPE,
   FROM_AGE              MONEY_TYPE,
   TO_AGE                MONEY_TYPE,
   PARAM_DESC            DESC_TYPE,
   COMMIT_FLAG           FLAG_TYPE,

   CREATE_DATE           DATE_TYPE,
   CREATE_BY             ID_TYPE,
   MODIFY_DATE           DATE_TYPE,
   MODIFY_BY             ID_TYPE
);

ALTER TABLE PARAMETERS ADD CONSTRAINT PARAMETER_ID_PK PRIMARY KEY (PARAM_ID);

CREATE GENERATOR PARAMETER_SEQ;
