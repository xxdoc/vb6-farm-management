
CREATE TABLE COSTPRM_ITEM
(
   COSTPRM_ITEM_ID       ID_TYPE NOT NULL,
   PARAM_ID              ID_TYPE NOT NULL,
   FOOD_ID               ID_TYPE NOT NULL,
   COST_RATE             MONEY_TYPE,

   CREATE_DATE           DATE_TYPE,
   CREATE_BY             ID_TYPE,
   MODIFY_DATE           DATE_TYPE,
   MODIFY_BY             ID_TYPE
);

ALTER TABLE COSTPRM_ITEM ADD CONSTRAINT COSTPRM_ITEM_ID_PK PRIMARY KEY (COSTPRM_ITEM_ID);
ALTER TABLE COSTPRM_ITEM ADD CONSTRAINT COSTPRM_ITEM_PARAM_ID_FK FOREIGN KEY (PARAM_ID) REFERENCES PARAMETERS;
ALTER TABLE COSTPRM_ITEM ADD CONSTRAINT COSTPRM_ITEM_FOOD_ID_FK FOREIGN KEY (FOOD_ID) REFERENCES PART_ITEM;

CREATE GENERATOR COSTPRM_ITEM_SEQ;
